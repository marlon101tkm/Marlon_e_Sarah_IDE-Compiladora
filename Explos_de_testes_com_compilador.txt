/* LOOP-> PARA */
vazio PRINCIPAL(){
	inteiro C;
	para(inteiro A = 0 ; A<10 ; A++){
		C = A;	
	}
}

/* LOOP-> PARA (composto) */
vazio PRINCIPAL(){
	inteiro C;
	para(inteiro A = 0 ; A<10 ; A++){
		para(inteiro B = 0 ; B<10 ; B++){
			C = A + B;	
		}
	}
}

/* LOOP-> FAZ */
vazio PRINCIPAL(){
	inteiro C;
	faz{
		C = C+1;
	}enquanto(C< 10);	
}

____________________________________________________________________________________________________________________________________

/* LOOP-> ENQUANTO */
vazio PRINCIPAL(){
	inteiro C;
	enquanto(C< 10){
	    C = C+1;
	}
}
